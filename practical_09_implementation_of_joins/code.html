USE hospital_db;

-- ────────── 1. INNER JOIN (Strict Overlapping Logic) ──────────
-- Extracting explicit mappings universally filtering exclusively intersecting connections natively.
SELECT p.name AS Patient_Name, a.appt_date, d.name AS Assignee_Doctor 
FROM PATIENT p
INNER JOIN APPOINTMENT a ON p.patient_id = a.patient_id
INNER JOIN DOCTOR d ON a.doctor_id = d.doctor_id
LIMIT 5;

-- ────────── 2. LEFT JOIN (Preserving Master Table Arrays) ──────────
-- Fetching absolutely ALL doctors regardless mapping entirely missing appointment logs reporting NULL.
SELECT d.name AS Doctor_Name, a.status AS Appointment_Status
FROM DOCTOR d
LEFT JOIN APPOINTMENT a ON d.doctor_id = a.doctor_id
ORDER BY Appointment_Status DESC LIMIT 5;

-- ────────── 3. RIGHT JOIN (Preserving Subsidiary Arrays) ──────────
-- Extracting matching doctors relying fundamentally generating output spanning complete Department tracking.
SELECT dep.dept_name, d.name AS Doctor
FROM DOCTOR d
RIGHT JOIN DEPARTMENT dep ON d.dept_id = dep.dept_id
LIMIT 5;

-- ────────── 4. FULL OUTER JOIN (Simulated Array Aggregation) ──────────
-- MySQL universally lacks FULL OUTER JOIN syntax. Simulating strict geometric union mapping overlaps natively.
SELECT d.name AS Doctor, a.appt_id
FROM DOCTOR d LEFT JOIN APPOINTMENT a ON d.doctor_id = a.doctor_id
UNION
SELECT d.name AS Doctor, a.appt_id
FROM DOCTOR d RIGHT JOIN APPOINTMENT a ON d.doctor_id = a.doctor_id
LIMIT 5;

-- ────────── 5. SELF JOIN (Recursive Internal Matrix Overlay) ──────────
-- Locating multiple arbitrary patients internally specifically generating same overlapping diagnosis parameters.
SELECT p1.name AS Patient_A, p2.name AS Patient_B, p1.disease AS Shared_Diagnosis
FROM PATIENT p1
INNER JOIN PATIENT p2 ON p1.disease = p2.disease
WHERE p1.patient_id <> p2.patient_id
ORDER BY Shared_Diagnosis ASC LIMIT 5;

-- ────────── 6. CROSS JOIN (Cartesian Algorithmic Multiplication) ──────────
-- Generating completely independent exact algorithmic permutation tables extracting combinations checking strictly scaling factors entirely.
-- Example: Match 10 Doctors * 5 Departments = 50 rows output.
SELECT d.name, dep.dept_name 
FROM DOCTOR d
CROSS JOIN DEPARTMENT dep LIMIT 3;
